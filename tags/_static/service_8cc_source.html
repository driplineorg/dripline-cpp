196  {

200  return true;
201  }

204

226

228  bool t_listen_error = false;

230  if( ! a_listener.listen_on_queue() )
231  {
232  t_listen_error = true;
233  this->cancel( RETURN_ERROR );
234  }
235  };
236 
237  for( async_map_t::iterator t_child_it = f_async_children.begin();
238  t_child_it != f_async_children.end();
239  ++t_child_it )

242  t_child_it->second->listener_thread() = std::thread( t_cancel_on_listen_error, std::ref(*t_child_it->second.get()) );

248  t_child_it != f_async_children.end();
249  ++t_child_it )

253  }
254 
255  f_receiver_thread.join();

258  {

260  }

262  {

264  }
265

413  bool service::listen_on_queue()
414  {
415  LINFO( dlog, "Listening for incoming messages on <" << f_name << ">" );
416 
417  while( ! is_canceled() )
418  {
419  amqp_envelope_ptr t_envelope;
420  core::post_listen_status t_post_listen_status = core::post_listen_status::unknown;
421  core::listen_for_message( t_envelope, t_post_listen_status, f_channel, f_consumer_tag, f_listen_timeout_ms );
422 
423  if( f_canceled.load() )
424  {
425  LDEBUG( dlog, "Service canceled" );
426  return true;
427  }
428 
429  if( t_post_listen_status == core::post_listen_status::timeout )
430  {
431  // we end up here every time the listen times out with no message received
432  continue;
433  }
434 
435  if( t_post_listen_status == core::post_listen_status::soft_error )
436  {
437  LWARN( dlog, "A soft error ocurred while listening for messages for <" << f_name << ">. The channel is still valid" );
438  continue;
439  }
440 
441  if( t_post_listen_status == core::post_listen_status::hard_error )
442  {
443  LERROR( dlog, "A hard error ocurred while listening for messages for <" << f_name << ">. The channel is no longer valid" );
444  return false;
445  }
446 
447  if( t_post_listen_status == core::post_listen_status::unknown )
448  {
449  LERROR( dlog, "An unknown status occurred while listening for messages for <" << f_name << ">" );
450  return false;
451  }
452 
453  // remaining status is core::post_listen_status::message_received
454 
455  f_status = status::processing;
456 
457  handle_message_chunk( t_envelope );
458 
459  if( f_canceled.load() )
460  {
461  LDEBUG( dlog, "Service <" << f_name << "> canceled" );
462  return true;
463  }
464 
465  f_status = status::listening;
466  }
467  return true;
