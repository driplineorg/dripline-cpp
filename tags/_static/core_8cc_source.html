251  sent_msg_pkg_ptr core::do_send( message_ptr_t a_message, const std::string& a_exchange, bool a_expect_reply ) const
252  {
253  // throws connection_error if it could not connect with the broker
254  // throws dripline_error if there's a problem with the exchange or creating the AMQP message object(s)
255  // returns the receive_reply package if the message was completely or partially sent
256  // the f_successful_send flag will be set accordingly: true if completely sent; false if partially sent
257  // if there was an error sending the message, that will be returned in f_send_error_message, which will be empty otherwise
258 
259  // lambda to create a string with the basic information about the send attempt
260  auto t_diagnostic_string_maker = [a_message, this]() -> std::string {
261  return std::string("Broker: ") + f_address +"\nPort: " + std::to_string(f_port) + "\nRouting Key: " + a_message->routing_key();
262  };
263 
264  if ( ! f_make_connection || core::s_offline )
265  {
266  throw a_message;
267  //throw dripline_error() << "cannot send reply when make_connection is false";
268  }
269 
270  amqp_channel_ptr t_channel = open_channel();
271  if( ! t_channel )
272  {
273  throw connection_error() << "Unable to open channel to send message\n" << t_diagnostic_string_maker();
274  }
275 
276  if( ! setup_exchange( t_channel, a_exchange ) )
277  {
278  throw dripline_error() << "Unable to setup the exchange <" << a_exchange << "> to send message\n" << t_diagnostic_string_maker();
279  }
280 
281  // create empty receive-reply object
282  sent_msg_pkg_ptr t_receive_reply = std::make_shared< sent_msg_pkg >();
283  std::unique_lock< std::mutex > t_rr_lock( t_receive_reply->f_mutex );
284 
285  if( a_expect_reply )
286  {
287  t_receive_reply->f_channel = t_channel;
288 
289  // create the reply-to queue, and bind the queue to the routing key over the given exchange
290  std::string t_reply_to = t_channel->DeclareQueue( "" );
291  t_channel->BindQueue( t_reply_to, a_exchange, t_reply_to );
292  // set the reply-to in the message because now we have the queue to which to reply
293  a_message->reply_to() = t_reply_to;
294 
295  // begin consuming on the reply-to queue
296  t_receive_reply->f_consumer_tag = t_channel->BasicConsume( t_reply_to );
297  LDEBUG( dlog, "Reply-to for request: " << t_reply_to );
298  LDEBUG( dlog, "Consumer tag for reply: " << t_receive_reply->f_consumer_tag );
299  }
300 
301  // convert the dripline::message object to an AMQP message
302  amqp_split_message_ptrs t_amqp_messages = a_message->create_amqp_messages( f_max_payload_size );
303  if( t_amqp_messages.empty() )
304  {
305  throw dripline_error() << "Unable to convert the dripline::message object to AMQP message(s) to be sent\n" << t_diagnostic_string_maker();
306  }
307 
308  try
309  {
310  LDEBUG( dlog, "Sending message to <" << a_message->routing_key() << ">" );
311  for( amqp_message_ptr& t_amqp_message : t_amqp_messages )
312  {
313  // send the message
314  // the first boolean argument is whether it's mandatory that the message be delivered to a queue.
315  // this is only the case for requests, where we expect something to be listening.
316  t_channel->BasicPublish( a_exchange, a_message->routing_key(), t_amqp_message, a_message->is_request(), false );
317  }
318  LDEBUG( dlog, "Message sent in " << t_amqp_messages.size() << " chunks" );
319  t_receive_reply->f_successful_send = true;
320  t_receive_reply->f_send_error_message.clear();
321  }
322  catch( AmqpClient::ConnectionClosedException& e )
323  {
324  LERROR( dlog, "Unable to send message because the connection is closed: " << e.what() );
325  throw connection_error() << "Unable to send message because the connection is closed: " << e.what() << '\n' << t_diagnostic_string_maker();
326  }
327  catch( AmqpClient::AmqpLibraryException& e )
328  {
329  LERROR( dlog, "AMQP error while sending message: " << e.what() );
330  t_receive_reply->f_successful_send = false;
331  t_receive_reply->f_send_error_message = std::string("AMQP error while sending message: ") + std::string(e.what()) + '\n' + t_diagnostic_string_maker();
332  }
333  catch( AmqpClient::MessageReturnedException& e )
334  {
335  LERROR( dlog, "Message was returned: " << e.what() );
336  t_receive_reply->f_successful_send = false;
337  t_receive_reply->f_send_error_message = std::string("Message was returned: ") + std::string(e.what()) + '\n' + t_diagnostic_string_maker();
338  }
339  catch( std::exception& e )
340  {
341  LERROR( dlog, "Error while sending message: " << e.what() );
342  t_receive_reply->f_successful_send = false;
343  t_receive_reply->f_send_error_message = std::string("Error while sending message: ") + std::string(e.what()) + '\n' + t_diagnostic_string_maker();
344  }
345 
346  return t_receive_reply;
347  }
